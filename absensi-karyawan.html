Setup Supabase Sync
1
Buka supabase.com → masuk ke project Anda → klik Settings → API
2
Copy Project URL dan anon public key, paste di bawah ini
3
Jalankan SQL ini di Supabase → SQL Editor untuk buat tabel:

create table employees ( id text primary key, name text not null, division text not null, position text, schedule_in text default '08:00', schedule_out text default '17:00', created_at timestamptz default now() ); create table attendance ( id text primary key, emp_id text references employees(id), emp_name text, division text, date text not null, time_in text, ts_in bigint, time_out text, ts_out bigint, status text default 'masuk', created_at timestamptz default now() ); alter table employees enable row level security; alter table attendance enable row level security; create policy "allow all" on employees for all using (true) with check (true); create policy "allow all" on attendance for all using (true) with check (true);
